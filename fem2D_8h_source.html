510 __global__ void kernel_femhat_reduce_data(double *data1, double *data2, int T1, int T2, int Tbegin1, int Tbegin2, int T1local, int T2local, int left_t1_idx, int left_t2_idx, double diff) {
511  int t2 = blockIdx.x*blockDim.x + threadIdx.x;
512 
513  if(t2 < T2local){
514  double center_t1 = (Tbegin2+t2)*diff;
515  double left_t1 = (Tbegin2+t2-1)*diff;
516  double right_t1 = (Tbegin2+t2+1)*diff;
517 
518  int id_counter = floor(left_t1) - left_t1_idx; /* first index in provided local t1 array */
519 
520  double phi_value; /* value of basis function */
521 
522  /* left part of hat function */
523  double mysum = 0.0;
524  int t1 = floor(left_t1);
525 
526  /* compute linear combination with coefficients given by basis functions */
527  while(t1 <= center_t1){
528  phi_value = (t1 - left_t1)/(center_t1 - left_t1);
529  mysum += phi_value*data1[id_counter];
530  t1 += 1;
531  id_counter += 1;
532  }
533 
534  /* right part of hat function */
535  while(t1 < right_t1){
536  phi_value = (t1 - right_t1)/(center_t1 - right_t1);
537  mysum += phi_value*data1[id_counter];
538  t1 += 1;
539  id_counter += 1;
540  }
