574 void Decomposition::permute_TRblocksize(Vec orig_Vec, Vec new_Vec, int blocksize, bool invert) const {
575  LOG_FUNC_BEGIN
576 
577  int Tbegin = get_Tbegin();
578  int Tend = get_Tend();
579  int Tlocal = get_Tlocal();
580  int Rbegin = get_Rbegin();
581  int Rlocal = get_Rlocal();
582 
583  int local_size = Tlocal*Rlocal*blocksize;
584 
585  IS orig_local_is;
586  IS new_local_is;
587 
588  Vec orig_local_Vec;
589  Vec new_local_Vec;
590 
591  /* prepare index set with local data */
592  int orig_local_arr[local_size];
593  int j = 0;
594  for(int t=Tbegin;t<Tend;t++){
595  for(int i=0;i<R;i++){
596  if(DDR_affiliation[i] == DDR_rank){
597  for(int k=0;k<blocksize;k++){
598  orig_local_arr[j*blocksize+k] = t*R*blocksize + i*blocksize + k;
599  }
600  j++;
601  }
602  }
603  }
604  TRYCXX( ISCreateGeneral(PETSC_COMM_WORLD, local_size, orig_local_arr, PETSC_COPY_VALUES,&orig_local_is) );
605 // TRYCXX( ISCreateGeneral(PETSC_COMM_WORLD, local_size, orig_local_arr, PETSC_OWN_POINTER,&orig_local_is) );
606 
607  TRYCXX( ISCreateStride(PETSC_COMM_WORLD, local_size, Tbegin*R*blocksize + Rbegin*blocksize, 1, &new_local_is) );
608 
609  /* get subvector with local values from original data */
610  TRYCXX( VecGetSubVector(new_Vec, new_local_is, &new_local_Vec) );
611  TRYCXX( VecGetSubVector(orig_Vec, orig_local_is, &orig_local_Vec) );
612 
613  /* copy values */
614  if(!invert){
615  TRYCXX( VecCopy(orig_local_Vec, new_local_Vec) );
616  } else {
617  TRYCXX( VecCopy(new_local_Vec, orig_local_Vec) );
618  }
619 
620  /* restore subvector with local values from original data */
621  TRYCXX( VecRestoreSubVector(new_Vec, new_local_is, &new_local_Vec) );
622  TRYCXX( VecRestoreSubVector(orig_Vec, orig_local_is, &orig_local_Vec) );
623 
624  /* destroy used stuff */
625  TRYCXX( ISDestroy(&orig_local_is) );
626  TRYCXX( ISDestroy(&new_local_is) );
627 
628  TRYCXX( PetscBarrier(NULL));
629 
630  LOG_FUNC_END
631 }
