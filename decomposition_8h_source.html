737 void Decomposition::permute_TRblocksize(Vec orig_Vec, Vec new_Vec, int blocksize, bool invert) const {
738  LOG_FUNC_BEGIN
739 
740  int Tbegin = get_Tbegin();
741  int Tend = get_Tend();
742  int Tlocal = get_Tlocal();
743  int Rbegin = get_Rbegin();
744  int Rlocal = get_Rlocal();
745 
746  int local_size = Tlocal*Rlocal*blocksize;
747 
748  IS orig_local_is;
749  IS new_local_is;
750 
751  Vec orig_local_Vec;
752  Vec new_local_Vec;
753 
754  /* prepare index set with local data */
755 // int orig_local_arr[local_size];
756  int *orig_local_arr;
757  orig_local_arr = new int [local_size];
758  int j = 0;
759 
760  for(int t=Tbegin;t<Tend;t++){
761  for(int i=0;i<R;i++){
762  if(DDR_affiliation[i] == DDR_rank){
763  for(int k=0;k<blocksize;k++){
764  orig_local_arr[j*blocksize+k] = t*R*blocksize + i*blocksize + k;
765  }
766  j++;
767  }
768  }
769  }
770 
771  TRYCXX( ISCreateGeneral(PETSC_COMM_WORLD, local_size, orig_local_arr, PETSC_COPY_VALUES,&orig_local_is) );
772 // TRYCXX( ISCreateGeneral(PETSC_COMM_WORLD, local_size, orig_local_arr, PETSC_OWN_POINTER,&orig_local_is) );
773 
774  TRYCXX( ISCreateStride(PETSC_COMM_WORLD, local_size, Tbegin*R*blocksize + Rbegin*blocksize, 1, &new_local_is) );
775 
776  /* get subvector with local values from original data */
777  TRYCXX( VecGetSubVector(new_Vec, new_local_is, &new_local_Vec) );
778  TRYCXX( VecGetSubVector(orig_Vec, orig_local_is, &orig_local_Vec) );
779 
780  /* copy values */
781  if(!invert){
782  TRYCXX( VecCopy(orig_local_Vec, new_local_Vec) );
783  } else {
784  TRYCXX( VecCopy(new_local_Vec, orig_local_Vec) );
785  }
786 
787  /* restore subvector with local values from original data */
788  TRYCXX( VecRestoreSubVector(new_Vec, new_local_is, &new_local_Vec) );
789  TRYCXX( VecRestoreSubVector(orig_Vec, orig_local_is, &orig_local_Vec) );
790 
791  /* destroy used stuff */
792  TRYCXX( ISDestroy(&orig_local_is) );
793  TRYCXX( ISDestroy(&new_local_is) );
794 
795  TRYCXX( PetscBarrier(NULL));
796 
797  LOG_FUNC_END
