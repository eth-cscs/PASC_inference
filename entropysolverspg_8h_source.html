451 
452  this->timer_solve.start();
453 
454  /* get dimensions */

457 
458  /* Anna knows the purpose of this number */
459  double eps = 0.0;
460 
461  /* prepare objects for Dlib */
462  column_vector Mom(Km);
463  column_vector starting_point(Km);
464 
465  /* stuff for PETSc to Dlib */
466  Vec moments_Vec = moments_data->get_vector();
467  Vec lambda_Vec = entropydata->get_lambda()->get_vector();
468  double *moments_arr;
469  double *lambda_arr;

471  TRYCXX( VecGetArray(lambda_Vec, &lambda_arr) );
472 
473  /* prepare lambda-functions for Dlib */
474  auto get_functions_obj_lambda = [&](const column_vector& x)->double { return get_functions_obj(x, Mom, eps);};
475  auto get_functions_grad_lambda = [&](const column_vector& x)->column_vector { return get_functions_grad(x, Mom, Km);};
476  auto get_functions_hess_lambda = [&](const column_vector& x)->dlib::matrix<double> { return get_functions_hess(x, Mom, Km);};
477 
478  /* through all clusters */
479  for(int k = 0; k < K; k++){
480  /* Mom: from PETSc vector to Dlib column_vector */
481  for(int km=0;km<Km;km++){
482  Mom(km) = moments_arr[k*Km+km];
483  }
484 
485  /* initial value form starting_point */
486  starting_point = 0.0;
487 
488  coutMaster << "k=" << k << ": running Dlib miracle" << std::endl;
489 
490  /* solve using Dlib magic */
491  dlib::find_min_box_constrained(dlib::newton_search_strategy(get_functions_hess_lambda),
492  dlib::objective_delta_stop_strategy(STOP_TOLERANCE).be_verbose(),
493  get_functions_obj_lambda, get_functions_grad_lambda, starting_point, -1e12, 1e12 );
494 
495 // coutMaster << "something computed" << std::endl;
496

498  for(int km=0;km<Km;km++){
499  lambda_arr[k*Km+km] = starting_point(km);
500  }
501 
502  } /* endfor through clusters */
503 
504  TRYCXX( VecRestoreArray(lambda_Vec, &lambda_arr) );
505  TRYCXX( VecRestoreArray(moments_Vec, &moments_arr) );

512 template<>
513 void EntropySolverSPG<PetscVector>::compute_moments_data() {
514  LOG_FUNC_BEGIN
515 
516  this->timer_compute_moments.start();
517 
518  Vec x_Vec = entropydata->get_x()->get_vector();
519  Vec x_power_Vec = x_power->get_vector();
520  Vec x_power_gammak_Vec = x_power_gammak->get_vector();
521  Vec gamma_Vec = entropydata->get_gamma()->get_vector();
522  Vec moments_Vec = moments_data->get_vector();
523 
524  Vec gammak_Vec;
525  IS gammak_is;
526 
527  TRYCXX( VecCopy(x_Vec,x_power_Vec) ); /* x^1 */
528 
529  double *moments_arr, mysum, gammaksum;
530  TRYCXX( VecGetArray(moments_Vec, &moments_arr) );
531  for(int km=0; km < entropydata->get_Km(); km++){
532 
533  for(int k=0;k<entropydata->get_K();k++){
534  /* get gammak */
535  this->entropydata->get_decomposition()->createIS_gammaK(&gammak_is, k);
536  TRYCXX( VecGetSubVector(gamma_Vec, gammak_is, &gammak_Vec) );
537 
538  /* compute x_power_gammak */
539  TRYCXX( VecPointwiseMult(x_power_gammak_Vec, gammak_Vec, x_power_Vec) ); /* x_power_gammak = x_power.*gammak */
540 
541  /* compute gammaksum */
542  TRYCXX( VecSum(gammak_Vec, &gammaksum) );
543  TRYCXX( VecSum(x_power_gammak_Vec, &mysum) );
544 
545  /* store computed moment */
546  if(gammaksum != 0){
547  moments_arr[k*this->entropydata->get_Km() + km] = mysum/gammaksum;
548  } else {
549  moments_arr[k*this->entropydata->get_Km() + km] = 0.0;
550  }
551 
552  TRYCXX( VecRestoreSubVector(gamma_Vec, gammak_is, &gammak_Vec) );
553  TRYCXX( ISDestroy(&gammak_is) );
554  }
555 
556  TRYCXX( VecPointwiseMult(x_power_Vec, x_Vec, x_power_Vec) ); /* x_power = x_power.*x */
557  }
558 
559  TRYCXX( VecRestoreArray(moments_Vec, &moments_arr) );
560 
561  this->timer_compute_moments.stop();
562 
563  LOG_FUNC_END
564 }

654  int Tlocal = entropydata->get_decomposition()->get_Tlocal();
655  int K = entropydata->get_K();
656  int Km = entropydata->get_Km();
657 
658  /* lambda vector for Dlib integration */
659  column_vector lambda(Km);
660  auto mom_function = [&](double x)->double { return gg(x, 0, lambda);};
661  double F_;
662 
663  /* update gamma_solver data - prepare new linear term */
664  /* theta includes all moments */
665  const double *lambda_arr;
666  TRYCXX( VecGetArrayRead(entropydata->get_lambda()->get_vector(), &lambda_arr) );
667 
668  const double *x_arr;
669  TRYCXX( VecGetArrayRead(entropydata->get_x()->get_vector(), &x_arr) );
670 
671  double *residuum_arr;
672  TRYCXX( VecGetArray(residuum->get_vector(), &residuum_arr) );
673 
674  double mysum, x_power;
675  for(int k=0;k<K;k++){
676  /* compute int_X exp(-sum lambda_j x^j) dx for this cluster
677  /* from arr to Dlib-vec */
678  for(int km=0;km<Km;km++){
679  lambda(km) = lambda_arr[k*Km+km];
680  }
681  F_ = dlib::integrate_function_adapt_simp(mom_function, -1.0, 1.0, 1e-10);
682  F_ = log(F_);
683 
684  for(int t=0;t<Tlocal;t++){
685  /* compute sum_{j=1}^{Km} lambda_j*x^j */
686  mysum = 0.0;
687  x_power = x_arr[t]; /* x^1 */
688  for(int km=0;km<Km;km++){
689  mysum += lambda_arr[k*Km+km]*x_power;
690  x_power *= x_arr[t]; /* x_power = x^(km+1) */
691  }
692 
693  residuum_arr[t*K + k] = mysum + F_;
694  }
695  }

697  /* coeffs of A_shared are updated via computation of Theta :) */
698 
699  /* restore arrays */

701  TRYCXX( VecRestoreArrayRead(entropydata->get_x()->get_vector(), &x_arr) );
702  TRYCXX( VecRestoreArrayRead(entropydata->get_lambda()->get_vector(), &lambda_arr) );
