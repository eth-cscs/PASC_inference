87 void PetscVectorWrapperComb::compute(const Vec &y, double init_scale){
88  if(DEBUG_MODE_PETSCVECTOR >= 100) std::cout << "(WrapperComb)FUNCTION: process(Vec,double)" << std::endl;
89 
90  int list_size = get_listsize();
91  PetscScalar *alphas;
92  Vec *vectors;
93  double scale = init_scale; /* = 0.0 if y=comb, = 1.0 if y+=comb */
94  double shift = 0.0;
95  int maxpy_length = 0;
96 
97  /* go throught the list:
98  * - if same vector => scale += coeff
99  * - if scalar (NULL Vec) => shift += coeff
100  * - otherwise prepare to array to maxpy
101  *
102  * afterwards
103  * y = scale*y (VecScale)
104  * y = y + shift (VecShift)
105  * y += alphas*vectors (VecMAXPY)
106  */
107 
108  /* allocate memory */
109  TRYCXX(PetscMalloc(sizeof(PetscScalar)*list_size,&alphas));
110  TRYCXX(PetscMalloc(sizeof(Vec)*list_size,&vectors));
111 
112  /* get array with coefficients and vectors */
113  std::list<PetscVectorWrapperCombNode>::iterator list_iter; /* iterator through list */
114  int j;
115 
116  /* set iterator to the begin of the list */
117  list_iter = comb_list.begin();
118 
119  /* go through the list and fill the vectors */
120  for(j=0;j<list_size;j++){
121  /* if Vec==NULL, then add to the shift */
122  if(list_iter->get_vector() == NULL){
123  shift += list_iter->get_coeff();
124  } else {
125  /* if same vector => scale += coeff */
126  if(list_iter->get_vector() == y){
127  scale+=list_iter->get_coeff();
128  } else {
129  /* otherwise prepare to maxpy-arrays */
130  alphas[maxpy_length] = list_iter->get_coeff();
131  vectors[maxpy_length] = list_iter->get_vector();
132 
133  maxpy_length += 1;
134  }
135  }
136 
137  if(j < list_size-1){
138  /* this is not the last element */
139  list_iter++;
140  }
141  }

144  if(DEBUG_MODE_PETSCVECTOR >= 99){
145  std::cout << " - linear combination:" << std::endl;
146  std::cout << " - scale: " << scale << std::endl;
147  std::cout << " - shift: " << shift << std::endl;
148  std::cout << " - maxpy: " << maxpy_length << std::endl;
149  }
