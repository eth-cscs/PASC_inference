210  Mat A = Abgs->get_petscmatrix();
211  double coeff = Abgs->get_coeff();
212  TRYCXX( MatScale(A, coeff) );
213  TRYCXX( MatAssemblyBegin(A,MAT_FINAL_ASSEMBLY) );
214  TRYCXX( MatAssemblyEnd(A,MAT_FINAL_ASSEMBLY) );

224  Vec c = fs_lineqbound->get_c();
225  Vec lb = fs_lineqbound->get_lb();
226 
227  Vec ub;
228  TRYCXX( VecDuplicate(lb,&ub) );
229  TRYCXX( VecSet(ub,1.0) );

233  TRYCXX( QPSetOperator(qp, A) ); /* set stiffness matrix */

487 template<class VectorBase>
488 void PermonSolver<VectorBase>::solve() {
489  LOG_FUNC_BEGIN
490 
491  this->timer_solve.start(); /* stop this timer in the end of solution */
492 
493  int it = -1;
494  int hessmult = -1;
495  double fx = std::numeric_limits<double>::max();
496 
497  /* call permon solver */
498  TRYCXX( QPSSolve(qps) );
499 
500  /* get the solution vector */
501  GeneralVector<PetscVector> *xg = dynamic_cast<GeneralVector<PetscVector> *>(qpdata->get_x());
502  Vec x = xg->get_vector();
503  TRYCXX( QPGetSolutionVector(qp, &x) );
504 
505  TRYCXX( QPSGetIterationNumber(qps, &it) );
506 
507  /* control the equality constraint */
508  SimplexFeasibleSet_LinEqBound<PetscVector> *fs_lineqbound = dynamic_cast<SimplexFeasibleSet_LinEqBound<PetscVector> *>(qpdata->get_feasibleset());
509  Mat B = fs_lineqbound->get_B();
510  Vec c = fs_lineqbound->get_c();
511  Vec v;
512  TRYCXX( VecDuplicate(c,&v) );
513  TRYCXX( MatMult(B,x,v) );
514  TRYCXX( VecAXPY(v,1.0,c) );
515 
516  double mynorm;
517  TRYCXX( VecNorm(v, NORM_2, &mynorm) );
518  coutMaster << "++++ norm(B*x - c) = " << mynorm << std::endl;
519 
520 
521  this->it_sum += it;
522  this->hessmult_sum += hessmult;
523  this->it_last = it;
524  this->hessmult_last = hessmult;
525 
526  this->fx = fx;
527  this->timer_solve.stop();
528 
529  /* write info to log file */
530  LOG_IT(it)
531  LOG_FX(fx)
