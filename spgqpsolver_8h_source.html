211 void SPGQPSolver<VectorBase>::solve() {
212  if(setting.debug_mode >= 100) std::cout << "(SPGQPSolver)FUNCTION: solve" << std::endl;
213 
214 MemoryCheck::test_temp(1);
215 
216  /* I don't want to write (*x) as a vector, therefore I define following pointer types */
217  typedef GeneralVector<VectorBase> (&pVector);
218  typedef GeneralMatrix<VectorBase> (&pMatrix);
219 
220  /* pointers to qpdata */
221  pMatrix A = *(qpdata->get_A());
222  pVector b = *(qpdata->get_b());
223  pVector x0 = *(qpdata->get_x0());
224 
225  /* pointer to solution */
226  pVector x = *(qpdata->get_x());
227 
228  /* auxiliary vectors */
229  pVector g = *(this->g); /* gradient */
230  pVector d = *(this->d); /* A-conjugate vector */
231  pVector Ad = *(this->Ad); /* A*p */
232 
233  int it = 0; /* number of iterations */
234  this->it = it;
235 
236  int hessmult = 0; /* number of hessian multiplications */
237 
238  double fx; /* function value */
239  SPGQPSolver_fs fs(setting.m); /* store function values for generalized Armijo condition */
240  double fx_max; /* max(fs) */
241  double xi, beta_bar, beta_hat, beta; /* for Armijo condition */
242  double dd; /* dot(d,d) */
243  double gd; /* dot(g,d) */
244  double dAd; /* dot(Ad,d) */
245  double alpha_bb; /* BB step-size */
246 
247  /* initial step-size */
248  alpha_bb = setting.alphainit;
249 
250  x = x0; /* set approximation as initial */
251  qpdata->get_feasibleset()->project(x); /* project initial approximation to feasible set */
252 
253  /* compute gradient, g = A*x-b */
254  g = A*x;
255  hessmult += 1; /* there was muliplication by A */
256  g -= b;
257 
258  /* compute function value */
259  fx = get_fx();
260  this->fx = fx;
261 
262  /* initialize fs */
263  fs.init(fx);
264 
265 MemoryCheck::test_temp(2);
266 
267 
268  /* main cycle */
269  while(it < setting.maxit){
270 
271  /* d = x - alpha_bb*g, see next step, it will be d = P(x - alpha_bb*g) - x */
272  d = x - alpha_bb*g;
273 
274 MemoryCheck::test_temp(3);
275 
276  /* d = P(d) */
277  qpdata->get_feasibleset()->project(d);
278 
279 MemoryCheck::test_temp(4);
280 
281  /* d = d - x */
282  d -= x;
283 
284  /* Ad = A*d */
285  Ad = A*d;
286  hessmult += 1; /* there was multiplication by A */
287 
288  /* dd = dot(d,d) */
289  /* dAd = dot(Ad,d) */
290  /* gd = dot(g,d) */
291  dd = dot(d,d);
292  dAd = dot(Ad,d);
293  gd = dot(g,d);
294 
295  /* stopping criteria */
296  if(dd < setting.eps){
297  break;
298  }
299 
300 MemoryCheck::test_temp(5);
301 
302  /* fx_max = max(fs) */
303  fx_max = fs.get_max();
304 
305  /* compute step-size from A-condition */
306  xi = (fx_max - fx)/dAd;
307  beta_bar = -gd/dAd;
308  beta_hat = setting.gamma*beta_bar + std::sqrt(setting.gamma*setting.gamma*beta_bar*beta_bar + 2*xi);
309 
310 MemoryCheck::test_temp(6);
311 
312  /* beta = min(sigma2,beta_hat) */
313  if(beta_hat < setting.sigma2){
314  beta = beta_hat;
315  } else {
316  beta = setting.sigma2;
317  }
318 
319  /* update approximation and gradient */
320  x += beta*d; /* x = x + beta*d */
321  g += beta*Ad; /* g = g + beta*Ad */
322 
323 MemoryCheck::test_temp(7);
324 
325  /* compute new function value using gradient and update fs list */
326  fx = get_fx();
327  fs.update(fx);
328 
329  /* update BB step-size */
330  alpha_bb = dd/dAd;
331 
332  /* print qpdata */
333  if(setting.debug_mode >= 10){

335  std::cout << "d: " << d << std::endl;

337  std::cout << "Ad: " << Ad << std::endl;
338 
339  }
