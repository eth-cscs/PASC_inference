579 /* solve the problem */
580 template<class VectorBase>
581 void SPGQPSolver<VectorBase>::solve() {
582  LOG_FUNC_BEGIN
583 
584  this->timer_solve.start(); /* stop this timer in the end of solution */
585 
586  /* I don't want to write (*x) as a vector, therefore I define following pointer types */
587  typedef GeneralVector<VectorBase> (&pVector);
588  typedef GeneralMatrix<VectorBase> (&pMatrix);
589 
590  /* pointers to qpdata */
591  pMatrix A = *(qpdata->get_A());
592  pVector b = *(qpdata->get_b());
593  pVector x0 = *(qpdata->get_x0());
594 
595  /* pointer to solution */
596  pVector x = *(qpdata->get_x());
597 
598  /* auxiliary vectors */
599  pVector g = *(this->g); /* gradient */
600  pVector d = *(this->d); /* A-conjugate vector */
601  pVector Ad = *(this->Ad); /* A*p */
602 
603  int it = 0; /* number of iterations */
604  int hessmult = 0; /* number of hessian multiplications */
605 
606  double fx; /* function value */
607  double fx_old; /* f(x_{it - 1}) */
608  SPG_fs fs(this->m); /* store function values for generalized Armijo condition */
609  double fx_max; /* max(fs) */
610  double xi, beta_bar, beta_hat, beta; /* for Armijo condition */
611  double dd; /* dot(d,d) */
612  double gd; /* dot(g,d) */
613  double dAd; /* dot(Ad,d) */
614  double alpha_bb; /* BB step-size */
615  double normb = norm(b); /* norm of linear term used in stopping criteria */
616 
617  /* initial step-size */
618  alpha_bb = this->alphainit;
619 
620  //TODO: temp!
621  x = x0; /* set approximation as initial */
622 
623  this->timer_projection.start();
624  qpdata->get_feasibleset()->project(x); /* project initial approximation to feasible set */
625  this->timer_projection.stop();
626 
627  /* compute gradient, g = A*x-b */
628  this->timer_matmult.start();
629  g = A*x;
630  hessmult += 1; /* there was muliplication by A */
631  this->timer_matmult.stop();
632 
633  //TODO: temp!!!
634  g -= b;
635 
636  /* initialize fs */
637  this->timer_fs.start();
638  fx = get_fx();
639  fx_old = std::numeric_limits<double>::max();
640  this->fx = fx;
641  fs.init(fx);
642  this->timer_fs.stop();
643 
644  /* main cycle */
645  while(it < this->maxit){
646  /* increase iteration counter */
647  it += 1;
648 
649  /* d = x - alpha_bb*g, see next step, it will be d = P(x - alpha_bb*g) - x */
650  this->timer_update.start();
651  d = x - alpha_bb*g;
652  this->timer_update.stop();
653 
654  /* d = P(d) */
655  this->timer_projection.start();
656  qpdata->get_feasibleset()->project(d);
657  this->timer_projection.stop();
658 
659  /* d = d - x */
660  this->timer_update.start();
661  d -= x;
662  this->timer_update.stop();
663 
664  /* Ad = A*d */
665  this->timer_matmult.start();
666  Ad = A*d;
667  hessmult += 1;
668  this->timer_matmult.stop();
669 
670  /* dd = dot(d,d); dAd = dot(Ad,d); gd = dot(g,d); */
671  this->timer_dot.start();
672  compute_dots(&dd, &dAd, &gd);
673  this->timer_dot.stop();
674 
675  /* fx_max = max(fs) */
676  this->timer_fs.start();
677  fx_max = fs.get_max();
678  this->timer_fs.stop();
679 
680  /* compute step-size from A-condition */
681  this->timer_stepsize.start();
682  xi = (fx_max - fx)/dAd;
683  beta_bar = -gd/dAd;
684  beta_hat = this->gamma*beta_bar + PetscSqrtReal(this->gamma*this->gamma*beta_bar*beta_bar + 2*xi);
685 
686  /* beta = max(sigma1,min(sigma2,beta_hat)) */
687  if(beta_hat < this->sigma1){
688  beta_hat = this->sigma1;
689  }
690 
691  if(beta_hat < this->sigma2){
692  beta = beta_hat;
693  } else {
694  beta = this->sigma2;
695  }
696  this->timer_stepsize.stop();
697 
698  /* update approximation and gradient */
699  this->timer_update.start();
700  x += beta*d; /* x = x + beta*d */
701  g += beta*Ad; /* g = g + beta*Ad */
702  this->timer_update.stop();
703 
704  /* compute new function value using gradient and update fs list */
705  this->timer_fs.start();
706  fx_old = fx;
707 // fx = get_fx(fx_old,beta,gd,dAd);
708  fx = get_fx();
709  fs.update(fx);
710  allbarrier();
711  this->timer_fs.stop();
712 
713  /* update BB step-size */
714  this->timer_stepsize.start();
715  alpha_bb = dd/dAd;
716  this->timer_stepsize.stop();
717 
718  this->gP = dd;
719 
720  /* print progress of algorithm */
721  if(debug_print_it){
722  coutMaster << "\033[33m it = \033[0m" << it;
723 
724  std::streamsize ss = std::cout.precision();
725  coutMaster << ", \t\033[36mfx = \033[0m" << std::setprecision(17) << fx << std::setprecision(ss);
726 
727  coutMaster << ", \t\033[36mgP = \033[0m" << this->gP;
728  coutMaster << ", \t\033[36mdd = \033[0m" << dd << std::endl;
729 
730  /* log function value */
731  LOG_FX(fx)
732 
733  }
734 
735  /* print qpdata */
736  if(debug_print_vectors){
737 // coutMaster << "x: " << x << std::endl;
738 // coutMaster << "d: " << d << std::endl;
739 // coutMaster << "g: " << g << std::endl;
740 // coutMaster << "Ad: " << Ad << std::endl;
741  }
742 
743  if(debug_print_scalars){
744  coutMaster << "\033[36m alpha_bb = \033[0m" << alpha_bb << ",";
745  coutMaster << "\033[36m dAd = \033[0m" << dAd << ",";
746  coutMaster << "\033[36m gd = \033[0m" << gd << std::endl;
747 
748  coutMaster << "\033[36m fx = \033[0m" << fx << ",";
749  coutMaster << "\033[36m fx_max = \033[0m" << fx_max << ",";
750  coutMaster << "\033[36m xi = \033[0m" << xi << std::endl;
751 
752  coutMaster << "\033[36m beta_bar = \033[0m" << beta_bar << ",";
753  coutMaster << "\033[36m beta_hat = \033[0m" << beta_hat << ",";
754  coutMaster << "\033[36m beta = \033[0m" << beta << std::endl;
755 
756  }
757 
758  /* stopping criteria */
759  if( this->stop_difff && abs(fx - fx_old) < this->eps){
760  break;
761  }
762  if(this->stop_normgp && dd < this->eps){
763  break;
764  }
765  if(this->stop_normgp_normb && dd < this->eps*normb){
766  break;
767  }
768  if(this->stop_Anormgp && dAd < this->eps){
769  break;
770  }
771  if(this->stop_Anormgp_normb && dAd < this->eps*normb){
772  break;
773  }
774 
775  /* monitor - export values of stopping criteria */
776  if(this->monitor && GlobalManager.get_rank() == 0){
777  //TODO: this could be done in a different way
778  std::ofstream myfile;
779  myfile.open("log/spgqpsolver_monitor.m", std::fstream::in | std::fstream::out | std::fstream::app);
780 
781  std::streamsize ss = myfile.precision();
782  myfile << std::setprecision(17);
783 
784  myfile << "fx(" << it << ") = " << fx << "; ";
785  myfile << "alpha_bb(" << it << ") = " << alpha_bb << "; ";
786  myfile << "beta(" << it << ") = " << beta << "; ";
787  myfile << "norm_difff(" << it << ") = " << abs(fx - fx_old) << "; ";
788  myfile << "norm_gp(" << it << ") = " << dd << "; ";
789  myfile << "norm_Agp(" << it << ") = " << dAd << "; ";
790  myfile << std::endl;
791

793  myfile.close();
794  }
795 
796  } /* main cycle end */
797 
798  this->it_sum += it;
799  this->hessmult_sum += hessmult;
800  this->it_last = it;
801  this->hessmult_last = hessmult;
802 
803  this->fx = fx;
804  this->timer_solve.stop();

808  LOG_FX(fx)
