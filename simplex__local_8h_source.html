263  if(t<T){ /* maybe we call more than T kernels */
264  int k;
265 
266  bool is_inside = true;
267  double sum = 0.0;
268 
269  /* control inequality constraints */
270  for(k = 0; k < K; k++){ // TODO: could be performed parallely
271  if(x[t*K+k] < 0.0){
272  is_inside = false;
273  }
274  sum += x[t*K+k];
275  }
276 
277  /* control equality constraints */
278  if(sum != 1){
279  is_inside = false;
280  }
281 
282  /* if given point is not inside the feasible domain, then do projection */
283  if(!is_inside){
284  int j,i;
285  /* compute sorted x_sub */
286  double *y = new double[K];
287  double sum_y;
288  for(k=0;k<K;k++){
289  y[k] = x[t*K+k];
290  }
291  sort_bubble(y,K);
292 
293  /* now perform analytical solution of projection problem */
294  double t_hat = 0.0;
295  i = K - 1;
296  double ti;
297 
298  while(i >= 1){
299  /* compute sum(y) */
300  sum_y = 0.0;
301  for(j=i;j<K;j++){ /* sum(y(i,n-1)) */
302  sum_y += y[j];
303  }
304 
305  ti = (sum_y - 1.0)/(double)(K-i);
306  if(ti >= y[i-1]){
307  t_hat = ti;
308  i = -1; /* break */
309  } else {
310  i = i - 1;
311  }
312  }
313 
314  if(i == 0){
315  t_hat = (sum-1.0)/(double)K; /* uses sum=sum(x_sub) */
316  }
317 
318  for(k = 0; k < K; k++){ // TODO: could be performed parallely
319  /* (*x_sub)(i) = max(*x_sub-t_hat,0); */
320  ti = x[t*K+k] - t_hat;
321  if(ti > 0.0){
322  x[t*K+k] = ti;
323  } else {
