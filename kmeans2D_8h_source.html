49  double L[4];
50  double r1, r2, r1n, r2n;
51 
52  double R, c, s;
53 
54  /* Compute normally distributed random numbers via Box-Muller */
55  r1 = rand()/(double)(RAND_MAX);
56  r1 = 1.-r1; /* to change from [0,1) to (0,1], which we need for the log */
57 
58  r2 = rand()/(double)(RAND_MAX);
59  R = sqrt(-2.*log(r1));
60  c = cos(2.*M_PI*r2);
61  s = sin(2.*M_PI*r2);
62 
63  /* compute normal distributed random values */
64  r1n = R*c;
65  r2n = R*s;
66 
67  /* choleski decomposition of SPD covariance matrix */
68  L[0] = sqrt(covariance[0]);
69  L[1] = 0;
70  L[2] = covariance[1]/L[0];
71  L[3] = sqrt(covariance[3] - L[2]*L[2]);
72 
73  /* compute output values */
74  /* y = L*randn(2,1) + mean */
75  *value1 = L[0]*r1n + L[1]*r2n + mu[0];
76  *value2 = L[2]*r1n + L[3]*r2n + mu[1];
