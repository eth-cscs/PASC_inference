136  * sum(x_sub) = 1
137  * x_sub >= 0

140 void SimplexFeasibleSet<VectorBase>::get_projection_sub(double *x_sub, int n){
141  int i;
142 
143  bool is_inside = true;
144  double sum = 0.0;
145 
146  /* control inequality constraints */
147  for(i = 0; i < n; i++){ // TODO: could be performed parallely
148  if(x_sub[i] < 0.0){
149  is_inside = false;
150  }
151  sum += x_sub[i];
152  }
153 
154  /* control equality constraints */
155  if(sum != 1){
156  is_inside = false;
157  }
158 
159 
160  /* if given point is not inside the feasible domain, then do projection */
161  if(!is_inside){
162  int j;
163  /* compute sorted x_sub */
164  double y[n], sum_y;
165  for(i=0;i<n;i++){
166  y[i] = x_sub[i];
167  }
168  sort_bubble(y,n);
169 
170  /* now perform analytical solution of projection problem */
171  double t_hat = 0.0;
172  i = n - 1;
173  double ti;
174 
175  while(i >= 1){
176  /* compute sum(y) */
177  sum_y = 0.0;
178  for(j=i;j<n;j++){ /* sum(y(i,n-1)) */
179  sum_y += y[j];
180  }
181 
182  ti = (sum_y - 1.0)/(double)(n-i);
183  if(ti >= y[i-1]){
184  t_hat = ti;
185  i = -1; /* break */
186  } else {
187  i = i - 1;
188  }
189  }
190 
191  if(i == 0){
192  t_hat = (sum-1.0)/(double)n; /* uses sum=sum(x_sub) */
193  }
194 
195  for(i = 0; i < n; i++){ // TODO: could be performed parallely
196  /* (*x_sub)(i) = max(*x_sub-t_hat,0); */
197  ti = x_sub[i] - t_hat;
198  if(ti > 0.0){
199  x_sub[i] = ti;
200  } else {
201  x_sub[i] = 0.0;
202  }
203  }
