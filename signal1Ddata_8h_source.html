375 double Signal1DData<VectorBase>::compute_abserr_reconstructed(GeneralVector<VectorBase> &solution) const {
376  LOG_FUNC_BEGIN
377 
378  double abserr;
379 
380  /* compute recovered signal */
381  Vec gammak_Vec;
382  IS gammak_is;
383 
384  Vec data_abserr_Vec;
385  TRYCXX( VecDuplicate(this->datavector->get_vector(), &data_abserr_Vec) );
386 
387  /* abserr = -solution */
388  TRYCXX( VecCopy(solution.get_vector(),data_abserr_Vec));
389  TRYCXX( VecScale(data_abserr_Vec,-1.0));
390 
391  double *theta_arr;
392  TRYCXX( VecGetArray(this->thetavector->get_vector(),&theta_arr) );
393 
394  int K = this->get_K();
395 
396  for(int k=0;k<K;k++){
397  /* get gammak */
398  this->decomposition->createIS_gammaK(&gammak_is, k);
399  TRYCXX( VecGetSubVector(this->gammavector->get_vector(), gammak_is, &gammak_Vec) );
400 
401  /* add to recovered image */
402  TRYCXX( VecAXPY(data_abserr_Vec, theta_arr[k], gammak_Vec) );
403 
404  TRYCXX( VecRestoreSubVector(this->gammavector->get_vector(), gammak_is, &gammak_Vec) );
405  TRYCXX( ISDestroy(&gammak_is) );
406  }
407 
408  TRYCXX( VecRestoreArray(this->thetavector->get_vector(),&theta_arr) );
409 
410  /* compute mean(abs(solution - data_recovered) */
411  TRYCXX( VecAbs(data_abserr_Vec) );
412  TRYCXX( VecSum(data_abserr_Vec, &abserr) );
413  int T = this->get_T();
414  abserr = abserr/(double)T;
415 
416  LOG_FUNC_END
417
