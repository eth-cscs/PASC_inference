126 template<class VectorBase>
127 void KmeansH1Model<VectorBase>::initialize_gammasolver(GeneralSolver **gammasolver, const TSData<VectorBase> *tsdata){
128  /* in this case, gamma problem is QP with simplex feasible set */
129 
130  /* create data */
131  gammadata = new QPData<VectorBase>();
132  gammadata->set_x0(tsdata->get_gammavector()); /* the initial approximation of QP problem is gammavector */
133  gammadata->set_x(tsdata->get_gammavector()); /* the solution of QP problem is gamma */
134  gammadata->set_b(new GeneralVector<VectorBase>(*gammadata->get_x0())); /* create new linear term of QP problem */
135  gammadata->set_A(new BlockDiagLaplaceExplicitMatrix<VectorBase>(*gammadata->get_x0(),this->K, this->penalty)); /* create new blockdiagonal matrix */
136  gammadata->set_feasibleset(new SimplexFeasibleSet<VectorBase>(this->T,this->K)); /* the feasible set of QP is simplex */
137 
138  /* create solver */
139  *gammasolver = new QPSolver<VectorBase>(*gammadata);
140 
141  /* generate random data to gamma */
142  gammadata->get_x0()->set_random();
143 
144  /* project random values to feasible set */
145  gammadata->get_feasibleset()->project(*gammadata->get_x0());
146 
147 }
