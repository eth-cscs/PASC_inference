194 /* solve the problem */
195 template<class VectorBase>
196 void MultiCGSolver<VectorBase>::solve() {
197  LOG_FUNC_BEGIN
198 
199  /* for each block prepare CG solver and solve the problem */
200  CGQPSolver<VectorBase> *cgsolver; /* cg solver for one block */
201  QPData<VectorBase> data_sub; /* data of inner cg solver */
202 
203  /* get number of blocks and blocks */
204  BlockDiagMatrix<VectorBase,LocalDenseMatrix<VectorBase> > *A = dynamic_cast<BlockDiagMatrix<VectorBase,LocalDenseMatrix<VectorBase> > *>(qpdata->get_A());
205  LocalDenseMatrix<VectorBase> **blocks = A->get_blocks();
206  LocalDenseMatrix<VectorBase> *A_sub;
207  int nmb_blocks = A->get_nmb_blocks();
208  int blocksize = A->get_blocksize();
209 
210  /* get vectors */
211  typedef GeneralVector<VectorBase> (&pVector);
212  pVector b = *(qpdata->get_b());
213  pVector x = *(qpdata->get_x());
214  pVector x0 = *(qpdata->get_x0());
215 
216  GeneralVector<VectorBase> b_sub(blocksize);
217  GeneralVector<VectorBase> x_sub(blocksize);
218  GeneralVector<VectorBase> x0_sub(blocksize);
219 
220  /* through blocks */
221  for(int i=0;i<nmb_blocks;i++){
222  /* get data for subproblem */
223  A_sub = blocks[i];
224  b_sub = 1*b(i*blocksize, (i+1)*blocksize - 1); // TODO: something is wrong with petscvector
225  x_sub = 1*x(i*blocksize, (i+1)*blocksize - 1); // TODO: something is wrong with petscvector
226  x0_sub = 1*x0(i*blocksize, (i+1)*blocksize - 1); // TODO: something is wrong with petscvector
227 
228  /* set data of subproblem to ... subproblem data :) */
229  data_sub.set_A(A_sub);
230  data_sub.set_b(&b_sub);
231  data_sub.set_x(&x_sub);
232  data_sub.set_x0(&x0_sub);
233 
234  /* create new instance of solver, during the constructor the new temp vectors are initialized */
235  cgsolver = new CGQPSolver<VectorBase>(data_sub);
236 
237  /* copy settings */
238 
239  /* solve this subproblem */
240  cgsolver->solve();
241 
242  /* update iteration counter */
243  if(cgsolver->get_fx() > this->fx) this->fx = cgsolver->get_fx();
244  if(cgsolver->get_it() > this->it_last) this->it_last = cgsolver->get_it();
245  if(cgsolver->get_hessmult() > this->hessmult_last) this->hessmult_last = cgsolver->get_hessmult();
246 
247  /* solution back to global vector */
248  x(i*blocksize, (i+1)*blocksize - 1) = x_sub;
249 
250  free(cgsolver);
251 
252  // TODO: deal with iteration counters (max?)
253  }
254 
255  /* write info to log file */
256  LOG_IT(this->it_last)
257  LOG_FX(this->fx)
258 
259  LOG_FUNC_END
260 }
