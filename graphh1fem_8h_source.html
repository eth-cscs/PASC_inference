350 
351  /* create data */
352  gammadata = new QPData<PetscVector>();
353 
354  gammadata->set_x0(tsdata->get_gammavector()); /* the initial approximation of QP problem is gammavector */
355  gammadata->set_x(tsdata->get_gammavector()); /* the solution of QP problem is gamma */
356  gammadata->set_b(new GeneralVector<PetscVector>(*gammadata->get_x0())); /* create new linear term of QP problem */
357 
358 // double coeff = (1.0/((double)(this->tsdata->get_R()*this->tsdata->get_T())))*this->epssqr;
359 // double coeff = (1.0/(sqrt((double)(this->tsdata->get_R()*this->tsdata->get_T()))))*this->epssqr;
360  double coeff = this->epssqr;
361 // double coeff = sqrt((double)(this->tsdata->get_R()*this->tsdata->get_T()))*this->epssqr;
362 // double coeff = this->tsdata->get_R()*this->tsdata->get_T()*this->epssqr;
363 
364  /* SPARSE */
365  if(usethetainpenalty){
366  /* use thetavector as a vector of coefficient for scaling blocks */
367  A_shared = new BlockGraphSparseMatrix<PetscVector>(*(tsdata->get_decomposition()), coeff, tsdata->get_thetavector() );
368  } else {
369  /* the vector of coefficient of blocks is set to NULL, therefore Theta will be not used to scale in penalisation */
370  A_shared = new BlockGraphSparseMatrix<PetscVector>(*(tsdata->get_decomposition()), coeff, NULL );
371  }
372 
373  gammadata->set_A(A_shared);
374 
375  /* generate random data to gamma */
376  gammadata->get_x0()->set_random();
377 
378  /* project random values to feasible set to be sure that initial approximation is feasible */
379 // gammadata->get_feasibleset()->project(*gammadata->get_x0());
380 
381  /* automatic choice of solver */
382  if(this->gammasolvertype == SOLVER_AUTO){
383  this->gammasolvertype = SOLVER_SPGQP;
384  }
385 
386  /* SPG-QP solver */
387  if(this->gammasolvertype == SOLVER_SPGQP){
388  /* the feasible set of QP is simplex */
389  gammadata->set_feasibleset(new SimplexFeasibleSet_Local<PetscVector>(tsdata->get_Tlocal()*tsdata->get_Rlocal(),tsdata->get_K()));
390 
391  /* create solver */
392  *gammasolver = new SPGQPSolver<PetscVector>(*gammadata);
393 
394  /* set stopping criteria based on the size of x (i.e. gamma) */
395 // (*gammasolver)->set_eps((double)(this->tsdata->get_R()*this->tsdata->get_T())*(*gammasolver)->get_eps());
396  }
397 
398  /* SPG-QP solver */
399  if(this->gammasolvertype == SOLVER_SPGQP_COEFF){
400  /* the feasible set of QP is simplex */
401  gammadata->set_feasibleset(new SimplexFeasibleSet_Local<PetscVector>(tsdata->get_Tlocal()*tsdata->get_Rlocal(),tsdata->get_K()));
402 
403  /* create solver */
404  *gammasolver = new SPGQPSolverC<PetscVector>(*gammadata);
405 
406  /* set stopping criteria based on the size of x (i.e. gamma) */
407 // (*gammasolver)->set_eps((double)(this->tsdata->get_R()*this->tsdata->get_T())*(*gammasolver)->get_eps());
408  }

507 template<>
508 void GraphH1FEMModel<PetscVector>::update_gammasolver(GeneralSolver *gammasolver){
509  LOG_FUNC_BEGIN
510 
511  int T = tsdata->get_T();
512  int Tlocal = tsdata->get_Tlocal();
513 
514  int R = tsdata->get_R();
515  int Rlocal = tsdata->get_Rlocal();
516 
517  int K = tsdata->get_K();
518 
519  /* update gamma_solver data - prepare new linear term */
520  const double *theta_arr;
521  TRYCXX( VecGetArrayRead(tsdata->get_thetavector()->get_vector(), &theta_arr) );
522 
523  const double *data_arr;
524  TRYCXX( VecGetArrayRead(tsdata->get_datavector()->get_vector(), &data_arr) );
525 
526  double *b_arr;
527  TRYCXX( VecGetArray(gammadata->get_b()->get_vector(), &b_arr) );
528 
529  double coeff = (-1.0)/((double)(R*T));
530 // double coeff = (-1.0)/(sqrt((double)(R*T)));
531 
532  for(int t=0;t<Tlocal;t++){
533  for(int r=0;r<Rlocal;r++){
534  for(int k=0;k<K;k++){
535  b_arr[t*K*Rlocal + r*K + k] = coeff*(data_arr[t*Rlocal+r] - theta_arr[k])*(data_arr[t*Rlocal+r] - theta_arr[k]);
536  }
537  }
538  }
539 
540  /* coeffs of A_shared are updated via computation of Theta :) */
541 
542  /* restore arrays */
543  TRYCXX( VecRestoreArray(gammadata->get_b()->get_vector(), &b_arr) );
544  TRYCXX( VecRestoreArrayRead(tsdata->get_datavector()->get_vector(), &data_arr) );
545  TRYCXX( VecRestoreArrayRead(tsdata->get_thetavector()->get_vector(), &theta_arr) );
546 
547  LOG_FUNC_END

550 /* update theta solver */
551 template<>
552 void GraphH1FEMModel<PetscVector>::update_thetasolver(GeneralSolver *thetasolver){
553  LOG_FUNC_BEGIN
554 
555  // TODO: if Theta is not in penalty term, this computation is completely WRONG! However, it doesn't matter if Theta is given
556 
557  Vec gamma_Vec = tsdata->get_gammavector()->get_vector();
558  Vec theta_Vec = tsdata->get_thetavector()->get_vector();
559  Vec data_Vec = tsdata->get_datavector()->get_vector();
560 
561  /* I will use A_shared with coefficients equal to 1, therefore I set Theta=1 */
562  TRYCXX( VecSet(tsdata->get_thetavector()->get_vector(),1.0) );
563  TRYCXX( VecAssemblyBegin(tsdata->get_thetavector()->get_vector()) );
564  TRYCXX( VecAssemblyEnd(tsdata->get_thetavector()->get_vector()) );
565 
566  /* now compute A*gamma */
567  Vec Agamma_Vec;
568  if(usethetainpenalty){
569  /* only if Theta is in penalty term */
570  *Agamma = (*A_shared)*(*(tsdata->get_gammavector()));
571  Agamma_Vec = Agamma->get_vector();
572  }
573 
574  /* subvectors */
575  Vec gammak_Vec;
576  Vec Agammak_Vec;
577  IS gammak_is;
578 
579  double gammakAgammak;
580  double gammakx;
581  double gammaksum;
582 
583  /* get arrays */
584  double *theta_arr;
585  TRYCXX( VecGetArray(theta_Vec,&theta_arr) );
586 
587  int K = tsdata->get_K();
588 
589 // double coeff = 1.0;
590  double coeff = 1.0/((double)(tsdata->get_R()*tsdata->get_T()));
591 
592  /* through clusters */
593  for(int k=0;k<K;k++){
594 
595  /* get gammak */
596  this->tsdata->get_decomposition()->createIS_gammaK(&gammak_is, k);
597  TRYCXX( VecGetSubVector(gamma_Vec, gammak_is, &gammak_Vec) );
598 
599  /* compute gammakAgammak */
600  if(usethetainpenalty){
601  /* only if Theta is in penalty term */
602  TRYCXX( VecGetSubVector(Agamma_Vec, gammak_is, &Agammak_Vec) );
603  TRYCXX( VecDot(gammak_Vec, Agammak_Vec, &gammakAgammak) );
604  }
605 
606  /* compute gammakx */
607  TRYCXX( VecDot(data_Vec, gammak_Vec, &gammakx) );
608 
609  /* compute gammaksum */
610  TRYCXX( VecSum(gammak_Vec, &gammaksum) );
611 
612  if(usethetainpenalty){
613  /* only if Theta is in penalty term */
614  if(coeff*gammaksum + 0.5*gammakAgammak != 0){
615  theta_arr[k] = (coeff*gammakx)/(coeff*gammaksum + 0.5*gammakAgammak);
616  } else {
617  theta_arr[k] = 0.0;
618  }
619  } else {
620  /* if Theta is not in penalty term, then the computation is based on kmeans */
621  if(gammaksum != 0){
622  theta_arr[k] = gammakx/gammaksum;
623  } else {
624  theta_arr[k] = 0.0;
625  }
626  }
627 
628  TRYCXX( VecRestoreSubVector(gamma_Vec, gammak_is, &gammak_Vec) );
629  if(usethetainpenalty){
630  /* only if Theta is in penalty term */
631  TRYCXX( VecRestoreSubVector(Agamma_Vec, gammak_is, &Agammak_Vec) );
632  }
633  TRYCXX( ISDestroy(&gammak_is) );
634  }
635 
636  /* restore arrays */
637  TRYCXX( VecRestoreArray(theta_Vec,&theta_arr) );
