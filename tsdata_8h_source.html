647 void TSData<PetscVector>::cutgamma() const{
648  LOG_FUNC_BEGIN
649 
650  int max_id;
651  double max_value;
652 
653  int K = get_K();
654  int gamma_t = decomposition->get_Tlocal()*decomposition->get_Rlocal();
655 
656  double *gamma_arr;
657  TRYCXX( VecGetArray(gammavector->get_vector(),&gamma_arr) );
658 
659  int t,k;
660  for(t = 0; t < gamma_t; t++){
661  /* find max value */
662  max_id = 0;
663  max_value = gamma_arr[t];
664  for(k = 1; k < K; k++){
665  if(gamma_arr[t*K + k] > max_value){
666  max_id = k;
667  max_value = gamma_arr[t*K + k];
668  }
669  }
670 
671  /* set new values */
672  for(k = 0; k < K; k++){
673  if(k == max_id){
674  gamma_arr[t*K + k] = 1.0;
675  } else {
676  gamma_arr[t*K + k] = 0.0;
677  }
678  }
679 
680 
681  }
682 
683  TRYCXX( VecRestoreArray(gammavector->get_vector(),&gamma_arr) );
684 
685  LOG_FUNC_END
686 }
