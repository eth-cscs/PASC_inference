82 
83  this->alpha = alpha;

87

89  int Tlocal = get_Tlocal();
90  int Tbegin = decomposition->get_Tbegin();
91  int Tend = decomposition->get_Tend();
92  int R = get_R();
93  int Rlocal = get_Rlocal();
94  int Rbegin = decomposition->get_Rbegin();
95  int Rend = decomposition->get_Rend();
96 
97  int* neighbor_nmbs = decomposition->get_graph()->get_neighbor_nmbs();
98  int **neightbor_ids = decomposition->get_graph()->get_neighbor_ids();
99 
100  /* create matrix */
101  TRYCXX( MatCreate(PETSC_COMM_WORLD,&A_petsc) );
102  TRYCXX( MatSetSizes(A_petsc,K*Rlocal*Tlocal,K*Rlocal*Tlocal,K*R*T,K*R*T) );
103 
104  #ifndef USE_CUDA
105  TRYCXX( MatSetType(A_petsc,MATMPIAIJ) );
106  #else
107  TRYCXX( MatSetType(A_petsc,MATAIJCUSPARSE) );
108  #endif
109 
110  /* compute preallocation of number of non-zero elements in matrix */
111  TRYCXX( MatMPIAIJSetPreallocation(A_petsc,3*(1+2*decomposition->get_graph()->get_m_max()),NULL,2*(decomposition->get_graph()->get_m_max()+1),NULL) );
112  TRYCXX( MatSeqAIJSetPreallocation(A_petsc,3*(1+2*decomposition->get_graph()->get_m_max()),NULL) );
113 
114  TRYCXX( MatSetFromOptions(A_petsc) );
115 
116  double coeff = 1.0;
117 
119  for(int k=0; k < K; k++){
120  for(int r=Rbegin; r < Rend; r++){
121  for(int t=Tbegin;t < Tend;t++){
122  int diag_idx = t*R*K + r*K + k;
123  int r_orig = decomposition->get_invPr(r);
124 
125  int Wsum;
126 
127  /* compute sum of W entries in row */
128  if(t == 0 || t == T-1){
129  if(T > 1){
130  Wsum = 2*neighbor_nmbs[r_orig]+2; /* +1 for diagonal block */
131 // Wsum = 2*neighbor_nmbs[r_orig]; /* +1 for diagonal block */
132  } else {
133  Wsum = neighbor_nmbs[r_orig];
134  }
135  } else {
136  if(T > 1){
137  Wsum = 3*neighbor_nmbs[r_orig]+4; /* +2 for diagonal block */
138 // Wsum = 3*neighbor_nmbs[r_orig]; /* +2 for diagonal block */
139  } else {
140  Wsum = (neighbor_nmbs[r_orig])+1; /* +1 for diagonal block */
141  }
142  }
143 
144  /* diagonal entry */
145  TRYCXX( MatSetValue(A_petsc, diag_idx, diag_idx, coeff*Wsum, INSERT_VALUES) );
146 
147  /* my nondiagonal entries */
148  if(T>1){
149  if(t > 0) {
150  /* t - 1 */
151  TRYCXX( MatSetValue(A_petsc, diag_idx, diag_idx-R*K, -2*coeff, INSERT_VALUES) );
152  }
153  if(t < T-1) {
154  /* t + 1 */
155  TRYCXX( MatSetValue(A_petsc, diag_idx, diag_idx+R*K, -2*coeff, INSERT_VALUES) );
156  }
157  }
158 
159  /* non-diagonal neighbor entries */
160  for(int neighbor=0;neighbor<neighbor_nmbs[r_orig];neighbor++){
161  int r_new = decomposition->get_Pr(neightbor_ids[r_orig][neighbor]);
162  int idx2 = t*R*K + r_new*K + k;
163 
164  TRYCXX( MatSetValue(A_petsc, diag_idx, idx2, -coeff, INSERT_VALUES) );
165 // TRYCXX( MatSetValue(A_petsc, idx2, diag_idx, -coeff, INSERT_VALUES) );
166  if(t > 0) {
167  TRYCXX( MatSetValue(A_petsc, diag_idx, idx2-R*K, -coeff, INSERT_VALUES) );
168  }
169  if(t < T-1) {
170  TRYCXX( MatSetValue(A_petsc, diag_idx, idx2+R*K, -coeff, INSERT_VALUES) );
171  }
172  }
173  } /* end T */
174 
175  } /* end R */
176 
177  } /* end K */
178 
179  /* finish all writting in matrix */
180  TRYCXX( PetscBarrier(NULL) );
181 
182  /* assemble matrix */
183  TRYCXX( MatAssemblyBegin(A_petsc,MAT_FINAL_ASSEMBLY) );
184  TRYCXX( MatAssemblyEnd(A_petsc,MAT_FINAL_ASSEMBLY) );
185  TRYCXX( PetscObjectSetName((PetscObject)A_petsc,"Hessian matrix") );
186 
187  LOG_FUNC_END
188 }
