336  LOG_FUNC_BEGIN
337 
338  this->timer_compute_moments.start();
339 
340  Vec x_Vec = entropydata->get_x()->get_vector();
341  Vec x_power_Vec = x_power->get_vector();

354

356  /* get gammak */
357  this->entropydata->get_decomposition()->createIS_gammaK(&gammak_is, k);
358  TRYCXX( VecGetSubVector(gamma_Vec, gammak_is, &gammak_Vec) );
359 
360  /* compute x_power_gammak */
361  TRYCXX( VecPointwiseMult(x_power_gammak_Vec, gammak_Vec, x_power_Vec) ); /* x_power_gammak = x_power.*gammak */
362 
363  /* compute gammaksum */
364  TRYCXX( VecSum(gammak_Vec, &gammaksum) );
365  TRYCXX( VecSum(x_power_gammak_Vec, &mysum) );
366 
367  /* store computed moment */
368  if(gammaksum != 0){
369  moments_arr[k*this->entropydata->get_Km() + km] = mysum/gammaksum;
370  } else {
371  moments_arr[k*this->entropydata->get_Km() + km] = 0.0;
372  }
373 
374  TRYCXX( VecRestoreSubVector(gamma_Vec, gammak_is, &gammak_Vec) );
375  TRYCXX( ISDestroy(&gammak_is) );
376  }
377 
378  TRYCXX( VecPointwiseMult(x_power_Vec, x_Vec, x_power_Vec) ); /* x_power = x_power.*x */
379  }
380 
381  TRYCXX( VecRestoreArray(moments_Vec, &moments_arr) );
382
