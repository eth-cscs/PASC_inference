611 template<class VectorBase>
612 void TaoSolver<VectorBase>::solve() {
613  LOG_FUNC_BEGIN
614 
615  this->timer_solve.start(); /* stop this timer in the end of solution */
616 
617  TRYCXX( TaoCreate(PETSC_COMM_WORLD, &taosolver) );
618  TRYCXX( TaoSetType(taosolver, TAOIPM) ); //TODO: is this the only one TAOsolver type?
619  TRYCXX( TaoSetInitialVector(taosolver, userctx->x0) ); /* set initial approximation */
620  if(this->use_upperbound){
621  TRYCXX( VecDuplicate(userctx->lb,&(userctx->ub)) );
622  TRYCXX( VecSet(userctx->ub,1.0) ); //TODO: destroy?
623  TRYCXX( TaoSetVariableBounds(taosolver, userctx->lb, userctx->ub) );
624  } else {
625  userctx->ub = PETSC_NULL;
626  TRYCXX( TaoSetVariableBounds(taosolver, userctx->lb, PETSC_NULL) );
627  }
628  TRYCXX( TaoSetObjectiveAndGradientRoutine(taosolver, FormFunctionGradient, (void*)userctx) );
629  TRYCXX( TaoSetHessianRoutine(taosolver,userctx->A,userctx->A,FormHessian,(void*)userctx) );
630  TRYCXX( TaoSetEqualityConstraintsRoutine(taosolver,userctx->cE,FormEqualityConstraints,(void*)userctx) );
