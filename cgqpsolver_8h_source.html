281 /* solve the problem */
282 template<class VectorBase>
283 void CGQPSolver<VectorBase>::solve() {
284  LOG_FUNC_BEGIN
285 
286  /* I don't want to write (*x) as a vector, therefore I define following pointer types */
287  typedef GeneralVector<VectorBase> (&pVector);
288  typedef GeneralMatrix<VectorBase> (&pMatrix);
289 
290  /* pointers to qpdata */
291  pMatrix A = *(this->qpdata->get_A());
292  pVector b = *(this->qpdata->get_b());
293  pVector x0 = *(this->qpdata->get_x0());
294 
295  /* pointer to solution */
296  pVector x = *(this->qpdata->get_x());
297 
298  /* auxiliary vectors */
299  pVector g = *(this->g); /* gradient */
300  pVector p = *(this->p); /* A-conjugate vector */
301  pVector Ap = *(this->Ap); /* A*p */
302 
303  x = x0; /* set approximation as initial */
304 
305  int it = 0; /* iteration counter */
306  int hessmult = 0; /* number of hessian multiplications */
307  double normg, alpha, beta, pAp, gg, gg_old;
308 
309  g = A*x; hessmult += 1; /* compute gradient */
310  g -= b;
311 
312  p = g; /* initial conjugate direction */
313 
314  gg = dot(g,g);
315 
316  normg = std::sqrt(gg);
317 
318  while(normg > this->eps && it < this->maxit){
319  /* compute new approximation */
320 
321  Ap = A*p; hessmult += 1;
322 
323  /* compute step-size */
324  pAp = dot(Ap,p);
325 
326  alpha = gg/pAp;
327 
328  /* set new approximation, x = x - alpha*p */
329  x -= alpha*p;
330 
331  /* compute gradient recursively, g = g - alpha*Ap */
332  g -= alpha*Ap;
333  gg_old = gg;
334  gg = dot(g,g);
335 
336  normg = std::sqrt(gg);
337 
338  /* compute new A-orthogonal vector, p = g + beta*p */
339  beta = gg/gg_old;
340  p *= beta;
341  p += g;
342 
343  if(this->debugmode >= 10){
344  coutMaster << "it " << it << ": ||g|| = " << normg << std::endl;
345  }
346 
347  if(this->debugmode >= 100){
348  coutMaster << "x = " << x << std::endl;
349  coutMaster << "g = " << g << std::endl;
350  coutMaster << "p = " << p << std::endl;
351  coutMaster << "Ap = " << Ap << std::endl;
352  coutMaster << "pAp = " << pAp << std::endl;
353  coutMaster << "alpha = " << alpha << std::endl;
354  coutMaster << "beta = " << beta << std::endl;
355  coutMaster << "gg = " << gg << std::endl;
356  coutMaster << "gg_old = " << gg_old << std::endl;
357  coutMaster << "normg = " << normg << std::endl;
358 
359  coutMaster << "------------------------------------" << std::endl;
360  }
361 
362 
363  it += 1;
364 
365  }
366 
367  /* print output */
368  if(this->debugmode >= 10){
369  coutMaster << "------------------------" << std::endl;
370  coutMaster << " it_cg = " << it << std::endl;
371  coutMaster << " norm_g = " << normg << std::endl;
372  coutMaster << " hessmult = " << hessmult << std::endl;
373  }
374 
375  this->it_sum += it;
376  this->hessmult_sum += hessmult;
377  this->it_last = it;
378  this->hessmult_last = hessmult;
379 
380  this->fx = normg; /* fx = norm(g) */
381 
382  /* write info to log file */
383  LOG_IT(it)
384  LOG_FX(this->fx)
385 
386  LOG_FUNC_END
387 }
