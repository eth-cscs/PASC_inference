631 template<class VectorBase>
632 void SPGQPSolverC<VectorBase>::solve() {
633  LOG_FUNC_BEGIN
634 
635  /* deal with coefficient */
636  double epssqr = qpdata->get_A()->get_coeff();
637  double epssqrinv;
638  if(epssqr == 0){
639  epssqrinv = 1.0; //std::numeric_limits<double>::max(); //TODO: are you sure?
640  } else {
641  epssqrinv = 1.0/epssqr;
642  }
643 
644  qpdata->get_A()->set_coeff(1.0);
645 
646  this->timer_solve.start(); /* stop this timer in the end of solution */
647 
648  /* I don't want to write (*x) as a vector, therefore I define following pointer types */
649  typedef GeneralVector<VectorBase> (&pVector);
650  typedef GeneralMatrix<VectorBase> (&pMatrix);
651 
652  /* pointers to qpdata */
653  pMatrix A = *(qpdata->get_A());
654  pVector b = *(qpdata->get_b());
655  pVector x0 = *(qpdata->get_x0());
656 
657  /* pointer to solution */
658  pVector x = *(qpdata->get_x());
659 
660  /* auxiliary vectors */
661  pVector g = *(this->g); /* gradient */
662  pVector d = *(this->d); /* A-conjugate vector */
663  pVector Ad = *(this->Ad); /* A*p */
664 
665  int it = 0; /* number of iterations */
666  int hessmult = 0; /* number of hessian multiplications */
667 
668  double fx; /* function value */
669  double fx_old; /* f(x_{it - 1}) */
670  SPGQPSolverC_fs fs(this->m); /* store function values for generalized Armijo condition */
671  double fx_max; /* max(fs) */
672  double xi, beta_bar, beta_hat, beta; /* for Armijo condition */
673  double dd; /* dot(d,d) */
674  double gd; /* dot(g,d) */
675  double dAd; /* dot(Ad,d) */
676  double alpha_bb; /* BB step-size */
677  double normb = norm(b); /* norm of linear term used in stopping criteria */
678 
679  /* initial step-size */
680  alpha_bb = this->alphainit;
681 
682  x = x0; /* set approximation as initial */
683 
684  this->timer_projection.start();
685  qpdata->get_feasibleset()->project(x); /* project initial approximation to feasible set */
686  this->timer_projection.stop();
687 
688  /* compute gradient, g = A*x-b */
689  this->timer_matmult.start();
690  g = A*x;
691  g *= epssqr;
692  hessmult += 1; /* there was muliplication by A */
693  this->timer_matmult.stop();
694  g -= b;
695 
696  /* initialize fs */
697  this->timer_fs.start();
698  qpdata->get_A()->set_coeff(epssqr);
699  fx = get_fx();
700  qpdata->get_A()->set_coeff(1.0);
701 
702  fx_old = std::numeric_limits<double>::max();
703  this->fx = fx;
704  fs.init(fx);
705  this->timer_fs.stop();
706 
707  /* main cycle */
708  while(it < this->maxit){
709  /* increase iteration counter */
710  it += 1;
711 
712  /* d = x - alpha_bb*g, see next step, it will be d = P(x - alpha_bb*g) - x */
713  this->timer_update.start();
714  d = x - epssqrinv*alpha_bb*g;
715  this->timer_update.stop();
716 
717  /* d = P(d) */
718  this->timer_projection.start();
719  qpdata->get_feasibleset()->project(d);
720  this->timer_projection.stop();
721 
722  /* d = d - x */
723  this->timer_update.start();
724  d -= x;
725  this->timer_update.stop();
726 
727  /* Ad = A*d */
728  this->timer_matmult.start();
729  Ad = A*d;
730  hessmult += 1;
731  this->timer_matmult.stop();
732 
733  this->timer_dot.start();
734 // dd = dot(d,d);
735 // dAd = dot(Ad,d);
736 // gd = dot(g,d);
737  compute_dots(&dd, &dAd, &gd);
738  this->timer_dot.stop();
739 
740  /* fx_max = max(fs) */
741  this->timer_fs.start();
742  fx_max = fs.get_max();
743  this->timer_fs.stop();
744 
745  /* compute step-size from A-condition */
746  this->timer_stepsize.start();
747  xi = (fx_max - fx)/dAd;
748  beta_bar = -gd/dAd;
749  beta_hat = this->gamma*beta_bar + PetscSqrtReal(this->gamma*this->gamma*beta_bar*beta_bar + 2*epssqr*xi);
750 
751  /* beta = max(sigma1,min(sigma2,beta_hat)) */
752  if(epssqrinv*beta_hat < this->sigma1){
753  beta_hat = this->sigma1;
754  }
755 
756  if(epssqrinv*beta_hat < this->sigma2){
757  beta = epssqrinv*beta_hat;
758  } else {
759  beta = this->sigma2;
760  }
761  this->timer_stepsize.stop();
762 
763  /* update approximation and gradient */
764  this->timer_update.start();
765  x += beta*d; /* x = x + beta*d */
766  g += epssqr*beta*Ad; /* g = g + beta*Ad */
767  this->timer_update.stop();
768 
769  /* compute new function value using gradient and update fs list */
770  this->timer_fs.start();
771  fx_old = fx;
772 // fx = get_fx(fx_old,beta,gd,dAd);
773 
774  qpdata->get_A()->set_coeff(epssqr);
775  fx = get_fx();
776  qpdata->get_A()->set_coeff(1.0);
777 
778  fs.update(fx);
779  this->timer_fs.stop();
780 
781  /* update BB step-size */
782  this->timer_stepsize.start();
783  alpha_bb = dd/dAd;
784  this->timer_stepsize.stop();
785 
786  this->gP = dd;
787 
788  /* print progress of algorithm */
789  if(debug_print_it){
790  coutMaster << "\033[33m it = \033[0m" << it;
791 
792  std::streamsize ss = std::cout.precision();
793  coutMaster << ", \t\033[36mfx = \033[0m" << std::setprecision(17) << fx << std::setprecision(ss);
794 
795  coutMaster << ", \t\033[36mgP = \033[0m" << this->gP;
796  coutMaster << ", \t\033[36mdd = \033[0m" << dd << std::endl;
797  }
798 
799  /* print qpdata */
800  if(debug_print_vectors){
801  coutMaster << "x: " << x << std::endl;
802  coutMaster << "d: " << d << std::endl;
803  coutMaster << "g: " << g << std::endl;
804  coutMaster << "Ad: " << Ad << std::endl;
805  }
806 
807  if(debug_print_scalars){
808  coutMaster << "\033[36m alpha_bb = \033[0m" << alpha_bb << ",";
809  coutMaster << "\033[36m dAd = \033[0m" << dAd << ",";
810  coutMaster << "\033[36m gd = \033[0m" << gd << std::endl;
811 
812  coutMaster << "\033[36m fx = \033[0m" << fx << ",";
813  coutMaster << "\033[36m fx_max = \033[0m" << fx_max << ",";
814  coutMaster << "\033[36m xi = \033[0m" << xi << std::endl;
815 
816  coutMaster << "\033[36m beta_bar = \033[0m" << beta_bar << ",";
817  coutMaster << "\033[36m beta_hat = \033[0m" << beta_hat << ",";
818  coutMaster << "\033[36m beta = \033[0m" << beta << std::endl;
819 
820  }
821 
822  /* stopping criteria */
823  if( this->stop_difff && std::abs(fx - fx_old) < this->eps){
824  break;
825  }
826  if(this->stop_normgp && dd < this->eps){
827  break;
828  }
829  if(this->stop_normgp_normb && dd < this->eps*normb){
830  break;
831  }
832  if(this->stop_Anormgp && dAd < this->eps){
833  break;
834  }
835  if(this->stop_Anormgp_normb && dAd < this->eps*normb){
836  break;
837  }
838 
839  } /* main cycle end */
840 
841  qpdata->get_A()->set_coeff(epssqr);
842 
843  this->it_sum += it;
844  this->hessmult_sum += hessmult;
845  this->it_last = it;
846  this->hessmult_last = hessmult;
847 
848  this->fx = fx;
849  this->timer_solve.stop();
850 
851  /* write info to log file */
852  LOG_IT(it)
853  LOG_FX(fx)
854 
855  LOG_FUNC_END
856 }
