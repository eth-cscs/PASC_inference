358  xadj[i] = counter;
359  for(int j=0;j<neighbor_nmbs[i];j++){
360  adjncy[counter] = neighbor_ids[i][j];
361  counter++;
362  }
363  }

365

367  int nWeights = 1; /* something with weights of graph, I really don't know, sorry */
368

371  NULL, NULL, NULL, &DD_size, NULL,
372  NULL, NULL, &objval, DD_affiliation);
373 
374  /* free aux stuff */
375  free(xadj);
376  free(adjncy);
377  /* --------------------- */
378 
379  /* compute local lengths and permutation of global indexes */
380  for(int i=0;i<DD_size;i++){ /* use DD_length as counters */
381  DD_lengths[i] = 0;
382  }

392  }

394  DD_permutation[i] += DD_ranges[DD_affiliation[i]]; /* shift local indexes to global */

398  }

427 void BGMGraph::process(double threshold) {
428  LOG_FUNC_BEGIN
429 
430  this->threshold = threshold;
431 
432  /* prepare array for number of neighbors */
433  neighbor_nmbs = (int*)malloc(n*sizeof(int));
434 
435  #pragma omp parallel for
436  for(int i=0;i<n;i++){
437  neighbor_nmbs[i] = 0;
438  }
439 
440  /* get local array and work with it */
441  const double *coordinates_arr;
442  TRYCXX( VecGetArrayRead(coordinates->get_vector(),&coordinates_arr) );
443 
444  /* go throught graph - compute number of neighbors */
445 // #pragma omp parallel for
446  for(int i=0;i<n;i++){
447  for(int j=i+1;j<n;j++){
448  if(compute_normsqr(coordinates_arr, i, j) < threshold*threshold){
449  neighbor_nmbs[i] += 1;
450  neighbor_nmbs[j] += 1;
451  }
452  }
453  }
454 
455  /* prepare storages for neightbors ids */
456  neighbor_ids = (int**)malloc(n*sizeof(int*));
457 
458  #pragma omp parallel for
459  for(int i=0;i<n;i++){
460  neighbor_ids[i] = (int*)malloc(neighbor_nmbs[i]*sizeof(int));
461  }
462 
463  /* go throught graph - fill indexes of neighbors */
464  int *counters;
465  counters = (int*)malloc(n*sizeof(int));
466 
467  #pragma omp parallel for
468  for(int i=0;i<n;i++){
469  counters[i] = 0;
470  }
471 
472 // #pragma omp parallel for // TODO: here is a problem, cannot be used, maybe because of couter arrays?
473  for(int i=0;i<n;i++){
474  for(int j=i+1;j<n;j++){
475  if(compute_normsqr(coordinates_arr, i, j) < threshold*threshold){
476  neighbor_ids[i][counters[i]] = j;
477  neighbor_ids[j][counters[j]] = i;
478 
479  counters[i] += 1;
480  counters[j] += 1;
481 
482  this->m += 1;
483  }
484  }
485 
486  /* compute m_max (max degree of vertex) */
487  if(neighbor_nmbs[i] > m_max){
488  this->m_max = neighbor_nmbs[i];
489  }
490  }
491  free(counters);
492 
493  /* restore array */
494  TRYCXX( VecRestoreArrayRead(coordinates->get_vector(),&coordinates_arr) );
495 
496  #ifdef USE_CUDA
497  /* copy data to gpu */
498  gpuErrchk( cudaMalloc((void **)&neighbor_nmbs_gpu, n*sizeof(int)) );
499  gpuErrchk( cudaMemcpy( neighbor_nmbs_gpu, neighbor_nmbs, n*sizeof(int), cudaMemcpyHostToDevice) );
500 
501  /* allocate pointers on CPU */
502  neighbor_ids_cpugpu = (int**)malloc(n*sizeof(int*));
503 
504  for(int i=0;i<n;i++){
505  int mysize = neighbor_nmbs[i];
506 
507  gpuErrchk( cudaMalloc((void **)&(neighbor_ids_cpugpu[i]), mysize*sizeof(int)) );
508  gpuErrchk( cudaMemcpy( neighbor_ids_cpugpu[i], neighbor_ids[i], mysize*sizeof(int), cudaMemcpyHostToDevice) );
509  }
510 
511  /* copy pointers to arrays from CPU to GPU */
512  gpuErrchk( cudaMalloc((void **)&neighbor_ids_gpu, n*sizeof(int*)) );
513  gpuErrchk( cudaMemcpy( neighbor_ids_gpu, neighbor_ids_cpugpu, n*sizeof(int*), cudaMemcpyHostToDevice) );
514 
515  gpuErrchk( cudaDeviceSynchronize() );
516  #endif
517 
518  processed = true;
519 
520  LOG_FUNC_END
521 }
