347  LOG_FUNC_BEGIN
348 
349  double fx = std::numeric_limits<double>::max();
350 
351  /* use computed gradient in this->g to compute function value */
352  double tempt;
353 
354  /* get PETSc specific stuff from general */
355  GeneralVector<PetscVector> *b_p = dynamic_cast<GeneralVector<PetscVector> *>(qpdata->get_b());
356  GeneralVector<PetscVector> *x_p = dynamic_cast<GeneralVector<PetscVector> *>(qpdata->get_x());
357  GeneralVector<PetscVector> *g_p = dynamic_cast<GeneralVector<PetscVector> *>(this->g);
358  GeneralVector<PetscVector> *temp_p = dynamic_cast<GeneralVector<PetscVector> *>(this->temp);
359 
360  Vec b_Vec = b_p->get_vector();
361  Vec x_Vec = x_p->get_vector();
362  Vec g_Vec = g_p->get_vector();
363  Vec temp_Vec = temp_p->get_vector();
364 
365  /* temp = g - b; tempt = dot(temp,x); */
366  TRYCXX( VecCopy(g_Vec,temp_Vec) );
367  TRYCXX( VecAXPY(temp_Vec, -1.0, b_Vec));
368  TRYCXX( VecDot(x_Vec,temp_Vec,&tempt));
369  allbarrier();
370 
371  fx = 0.5*tempt;
372 
373  LOG_FUNC_END
374  return fx;
375 }
